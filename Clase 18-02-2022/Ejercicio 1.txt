//Leer 3 nÃºmeros y mostrar en orden creciente
//INPUT: 3 -5 2
//OUTPUT: -5 2 3
main(){
	int a, b, c;
	read(a, b, c);
	if(a > b){
		int aux <- a;
		a <- b;
		b <- aux;
	}
	if(a > c){
		int aux <- a;
		a <- c;
		c <- aux;
	}
	if(b > c){
		int aux <- b;
		b <- c;
		c <- aux;
	}
	print(a, b, c);
}
